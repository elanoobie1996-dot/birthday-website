    <script>
        // 1. Start Experience Logic
        function startExperience() {
            document.getElementById('intro-overlay').style.opacity = '0';
            setTimeout(() => {
                document.getElementById('intro-overlay').style.display = 'none';
                document.getElementById('main-content').style.display = 'block';
                document.body.style.overflowY = 'auto';
                document.body.style.overflowX = 'hidden';
                
                // Initialize first animation
                revealSections();
                
                // Play music (optional)
                const music = document.getElementById('bgMusic');
                music.volume = 0.3;
                music.play().catch(() => console.log("Autoplay blocked by browser."));
            }, 1500);
        }

        // 2. Custom Cursor Tracking
        const cursor = document.getElementById('cursor');
        const follower = document.getElementById('cursor-follower');
        
        document.addEventListener('mousemove', (e) => {
            cursor.style.left = e.clientX + 'px';
            cursor.style.top = e.clientY + 'px';
            
            setTimeout(() => {
                follower.style.left = (e.clientX - 16) + 'px';
                follower.style.top = (e.clientY - 16) + 'px';
            }, 50);
        });

        document.addEventListener('mousedown', () => {
            follower.style.transform = 'scale(0.8)';
            cursor.style.transform = 'scale(1.5)';
        });

        document.addEventListener('mouseup', () => {
            follower.style.transform = 'scale(1)';
            cursor.style.transform = 'scale(1)';
        });

        // 3. Particle System Background
        const canvas = document.getElementById('particleCanvas');
        const ctx = canvas.getContext('2d');
        let particles = [];

        function resize() {
            canvas.width = window.innerWidth;
            canvas.height = window.innerHeight;
        }

        window.addEventListener('resize', resize);
        resize();

        class Particle {
            constructor() {
                this.x = Math.random() * canvas.width;
                this.y = Math.random() * canvas.height;
                this.size = Math.random() * 2 + 0.5;
                this.speedX = Math.random() * 1 - 0.5;
                this.speedY = Math.random() * 1 - 0.5;
                this.opacity = Math.random() * 0.5 + 0.2;
            }
            update() {
                this.x += this.speedX;
                this.y += this.speedY;
                if (this.x > canvas.width) this.x = 0;
                if (this.x < 0) this.x = canvas.width;
                if (this.y > canvas.height) this.y = 0;
                if (this.y < 0) this.y = canvas.height;
            }
            draw() {
                ctx.fillStyle = `rgba(224, 176, 255, ${this.opacity})`;
                ctx.beginPath();
                ctx.arc(this.x, this.y, this.size, 0, Math.PI * 2);
                ctx.fill();
            }
        }

        function initParticles() {
            particles = [];
            for (let i = 0; i < 100; i++) {
                particles.push(new Particle());
            }
        }

        function animateParticles() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            particles.forEach(p => {
                p.update();
                p.draw();
            });
            requestAnimationFrame(animateParticles);
        }

        initParticles();
        animateParticles();

        // 4. Scroll Reveal Logic
        function revealSections() {
            const cards = document.querySelectorAll('.content-card');
            cards.forEach(card => {
                const rect = card.getBoundingClientRect();
                if (rect.top < window.innerHeight * 0.8) {
                    card.classList.add('active');
                }
            });
        }

        window.addEventListener('scroll', revealSections);

        // 5. Interactive Features
        function blowCandle() {
            const candle = document.getElementById('birthdayCandle');
            const status = document.getElementById('wish-status');
            candle.classList.add('blown');
            candle.innerHTML = '✨';
            status.style.opacity = '1';
            
            // Trigger Confetti Effect (Simple Emoji burst)
            for(let i=0; i<15; i++) {
                createConfetti();
            }
        }

        function createConfetti() {
            const conf = document.createElement('div');
            conf.innerText = '🌸';
            conf.style.position = 'fixed';
            conf.style.left = '50%';
            conf.style.top = '50%';
            conf.style.zIndex = '100';
            conf.style.pointerEvents = 'none';
            document.body.appendChild(conf);

            const destinationX = (Math.random() - 0.5) * 500;
            const destinationY = (Math.random() - 0.5) * 500;

            conf.animate([
                { transform: 'translate(0, 0) scale(1)', opacity: 1 },
                { transform: `translate(${destinationX}px, ${destinationY}px) scale(0)`, opacity: 0 }
            ], {
                duration: 1500,
                easing: 'cubic-bezier(0, .9, .57, 1)'
            }).onfinish = () => conf.remove();
        }

        function restart() {
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }
